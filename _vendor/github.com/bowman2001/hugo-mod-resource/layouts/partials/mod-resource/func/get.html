{{- $name := "mod-resource/func/get" }} {{- /* Initialize local variables with the values from the context map */}} {{- $url := .url }} {{- $ctx := .ctx }} {{- $errorLevel := .errorLevel }} {{- $errorMsg := printf "Unable to resolve the resource %q" $url.String }} {{- $r := "" }} {{- if $url.IsAbs }} {{- /* Remote */}} {{- with resources.GetRemote $url.String }} {{- with .Err }} {{- $msg := print $errorMsg ". " . }} {{- partial "md-error-msg" (dict "caller" $name "ctx" $ctx "errorMsg" $msg "errorLevel" $errorLevel ) }} {{- else }} {{- $r = . }} {{ end }} {{- else }} {{- partial "md-error-msg" (dict "caller" $name "ctx" $ctx "errorMsg" $errorMsg "errorLevel" $errorLevel) }} {{- end }} {{- else }} {{- $path := (strings.TrimPrefix "./" $url.Path ) }} {{- with $ctx.Page.Resources.GetMatch (print "*" $path "*") }} {{- /* Page */}} {{- $r = . }} {{- else }} {{- $args := dict "path" $path "ctx" $ctx }} {{- $r = partialCached "mod-resource/func/helper/page-fallback" $args $args }} {{- end }} {{- if not $r }} {{- with resources.Get $url.Path }} {{- /* Site */}} {{- $r = . }} {{- else }} {{- /* $r is still empty */}} {{- partial "md-error-msg" (dict "caller" $name "ctx" $ctx "errorMsg" $errorMsg "errorLevel" $errorLevel) }} {{- end }} {{- end }} {{- end }} {{- return $r }}
